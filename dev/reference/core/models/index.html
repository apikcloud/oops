    @classmethod
def from_string(cls, output: str, sep: str = ";") -> "CommitInfo":
    """ "--pretty=format:%h;%an;%ae;%ad;%s"
    1. sha
    2. author name
    3. author email
    4. date (ISO 8601 format)
    5. commit message
    """
    sha, author, email, date_str, message = output.split(sep, 4)
    commit_date = datetime.fromisoformat(date_str)
    return cls(
        sha=sha,
        author=author,
        email=email,
        date=commit_date,
        message=message,
    )
